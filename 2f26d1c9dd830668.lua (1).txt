-- Helper function to format time
local function formatTime(seconds)
    local hours = math.floor(seconds / 3600)
    local minutes = math.floor((seconds % 3600) / 60)
    local secs = math.floor(seconds % 60)
    return string.format("%02d:%02d:%02d", hours, minutes, secs)
end

-- Helper function to claim an egg
local function claimEgg(eggId, position, eggNumber)
    if not HumanoidRootPart or not HumanoidRootPart.Parent then
        warn("Character lost, waiting for respawn...")
        Character = LocalPlayer.CharacterAdded:Wait()
        HumanoidRootPart = Character:WaitForChild("HumanoidRootPart")
    end

    wait(CONFIG.WAIT_BETWEEN_EGGS)

    -- Try to claim the egg

    end)

    if success then
        print("✓ " .. eggId .. " (" .. eggNumber .. "/" .. #EGGS .. ")")
    else
        warn("✗ Failed to claim " .. eggId)
    end

    wait(CONFIG.WAIT_BETWEEN_LOCATIONS)
end
